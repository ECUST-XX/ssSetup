#!/bin/sh
[ $PASS = "**NULL**" ] && PASS=`601qaz106`
[ $PORT = "**NULL**" ] && PORT=`9621`

{
	"server":"0.0.0.0",
	"server_port":8388,

	"password":"${PASS}",
	"timeout":600,
	"method":"aes-256-cfb"

EOF
